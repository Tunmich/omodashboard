        self.rug_threshold = -5.0   # % decrease to sell

    def fetch_pairs(self):
        try:

            response.raise_for_status()
            data = response.json()

            logging.error(f"Error fetching pairs: {e}")
            return []

                return True
        except:
            pass
        return False
